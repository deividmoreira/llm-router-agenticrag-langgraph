# Projeto 7 - Sistema Web de Suporte Técnico Automatizado com Agentic RAG e LLM Routing

# Para executar a app:

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsaragp7 python=3.12

# Ative o ambiente:

conda activate dsaragp7 (ou: source activate dsaragp7)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute: 

python dsa_p7_setup_rag.py
streamlit run dsa_p7_streamlit_app.py

# Perguntas de exemplo:

# Meu wifi está desconectando, o que fazer?
# Não estou conseguindo instalar o Anaconda Python.

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsaragp7 --all
